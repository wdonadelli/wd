function runList(x) {
	x.style.backgroundColor = null;
	if (x.textContent === A) {
		x.textContent = Item 1;
		x.style.backgroundColor = #FF0000;
	} else if (x.textContent === B) {
		x.textContent = Item 2;
		x.style.backgroundColor = #00FF00;
	} else if (x.textContent === C) {
		x.textContent = Item 3;
		x.style.backgroundColor = #0000FF;
	} else if (x.textContent === Item 1) {
		x.textContent = A;
	} else if (x.textContent === Item 2) {
		x.textContent = B;
	} else if (x.textContent === Item 3) {
		x.textContent = C;
	}
	return;
}

var myObject = wd$(#List1 > ul > li);

myObject.run(runList);

var myObject = wd$(#Handler button)
			
function myMouseOver() {
	myObject.item().textContent = "Mouse Over";
	return;
}

function myMouseOut() {
	myObject.item().textContent = "Mouse Out";
	return;
}

function myMouseclick() {
	myObject.item().textContent = "Mouse Click";
	return;
}

myObject.handler({
	onmouseover: myMouseOver,
	mouseout: myMouseOut,
	OnClICk: myMouseclick
});
